<script>
$(function(){
var $container = $('#container');
 
$container.isotope({
itemSelector : '.story'
});
 
var $optionSets = $('#sorting .option-set'),
$optionLinks = $optionSets.find('a');
 
$optionLinks.click(function(){
var $this = $(this);
// don't proceed if already selected
if ( $this.hasClass('selected') ) {
return false;
}

var $optionSet = $this.parents('.option-set');
$optionSet.find('.selected').removeClass('selected');
$this.addClass('selected');
 
// make option object dynamically, i.e. { filter: '.my-filter-class' }
var options = {},
key = $optionSet.attr('data-option-key'),
value = $this.attr('data-option-filter');
// parse 'false' as false boolean
value = value === 'false' ? false : value;
options[ key ] = value;
if ( key === 'layoutMode' && typeof changeLayoutMode === 'function' ) {
// changes in layout modes need extra logic
changeLayoutMode( $this, options )
} else {
// otherwise, apply new options
$container.isotope( options );
}
 
return false;
});
 
});
</script>

<script>
imagesLoaded(document.querySelector('body'), function() {


/** variables: **/
let atInvrtBtt = document.querySelector("#at_inverter_cores");
let atButtonOri = document.querySelector("#at_inverter_cores .button-title.ori");
let atButtonAlt = document.querySelector("#at_inverter_cores .button-title.alt");
var colorModeCookie;

if(!location.hostname.startsWith("assets.txmblr")) {
        colorModeCookie = Cookies.get('themeColorMode');
} else {
        colorModeCookie = 0;
}; // if
    
/** invert colors function: **/
function invertColors(){

    if(colorModeCookie == 0) {
        atButtonOri.style.display="inline-block";
        atButtonAlt.style.display="none";
        document.querySelector("body").classList.remove("body-inverter");
        document.querySelectorAll("#menu, #sidebar, #container, #credit, #cookies-notice, img, figure, iframe").forEach(function(item){item.classList.remove("inverter");});
        document.querySelectorAll("figure img, .photoset figure").forEach(function(item){item.classList.remove("naoinverter");});
    } else {
        atButtonOri.style.display="none";
        atButtonAlt.style.display="inline-block";
        document.querySelector("body").classList.add("body-inverter");
        document.querySelectorAll("#menu, #sidebar, #container, #credit, #cookies-notice, img, figure, iframe").forEach(function(item){item.classList.add("inverter");});
        document.querySelectorAll("figure img, .photoset figure").forEach(function(item){item.classList.add("naoinverter");});
    }; // if
        
}; invertColors();

/** button on click: **/
atInvrtBtt.onclick = function(){
    
    /** switch cookie value **/
    if(!location.hostname.startsWith("assets.txmblr")) {
        if(colorModeCookie == 0) {
            Cookies.set('themeColorMode', '1', {expires: 180, sameSite: 'strict'});
            colorModeCookie = Cookies.get('themeColorMode');
        } else {
            Cookies.set('themeColorMode', '0', {expires: 180, sameSite: 'strict'});
            colorModeCookie = Cookies.get('themeColorMode');
        }; // if
    } else {
        if(colorModeCookie == 0) {
            colorModeCookie = 1;
        } else {
            colorModeCookie = 0;
        }; // if
    }; // if
    
    /** invert colors: **/
    invertColors();
    
}; // onclick

/** cookies message: **/
var cookiesConsent = 0;
    
// if not on Tumblr's Preview:
if(!location.hostname.startsWith("assets.txmblr")) {
    cookiesConsent = Cookies.get('at1cookiesConsent');
}; // if
    
function checkCookiesConsent() {
    if(cookiesConsent) {document.querySelector("#cookies-notice").style.display="none"};
}; checkCookiesConsent();

    } else {cookiesConsent = 1;}; // if
